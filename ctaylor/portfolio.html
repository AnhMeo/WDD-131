Shopping Cart

This is a shopping cart project that I made using python. It showcases the use of for loops and while loops, while also synchronizing lists.


                menuSelection = ''
                cartItems = []
                cartPrices = []
                total = 0

                while menuSelection != '5':
                    print('''Please select one of the following: 
                        1. Add item
                        2. View cart
                        3. Remove item
                        4. Compute total
                        5. Quit''')
                    menuSelection = input('Please enter an action: ')
                    if menuSelection == '1':
                        newItem = input('Please enter the name of the item you want to add: ')
                        newPrice = float(input(f'what is the price of {newItem}? '))
                        cartItems.append(newItem)
                        cartPrices.append(newPrice)
                        print()
                    elif menuSelection == '2':
                        print('The contents of the shopping cart are: ')
                        for i in range(len(cartItems)):
                            item = cartItems[i]
                            price = cartPrices[i]
                            print(f'{i+1}. {item} - {price}')
                            print()
                    elif menuSelection == '3':
                        removeItem = int(input('Which item would you like to remove? '))
                        removeIndex = removeItem - 1
                        listLength = len(cartItems)
                        if removeItem > listLength:
                            print('Sorry, that is not a valid item number.')
                            print()
                        else:
                            cartItems.pop(removeIndex)
                            cartPrices.pop(removeIndex)
                            print('Item removed.')
                            print()
                    elif menuSelection == '4':
                        for i in cartPrices:
                            total += i 
                        print(f'The total of all the items in the cart is ${total}.')
                        print()

Wordle Project

This is a coding project that allows you to play a game that is similar to Wordle


                import random

                play = 'yes'

                while play == 'yes':
                    secrets = ['vietnam', 'japan', 'america', 'philippines', 'canada', 'thailand', 'korea', 'spain', 'china', 'mexico', 'italy', 'france', 'cambodia', 'laos', 'singapore', 'malaysia', 'portugal', 'egypt', 'tunisia', 'argentina', 'brazil', 'venezuela', 'croatia', 'cyprus', 'iceland', 'ireland', 'denmark', 'saudi arabia', 'qatar']
                    secret = random.choice(secrets)
                    secretNum = len(secret)
                    guesses = 0
                    print('Welcome to the word guessing game!')

                    print('Here is your hint: ')
                    for i in range(secretNum):
                        print('_ ', end='')
                    print()
                    print()
                    guess = input('Enter your guess: ')
                    guesses = guesses + 1
                    while(guess != secret or len(guess) != secretNum):
                        if len(guess) != secretNum:
                            print('The number of letters must be ' + str(secretNum) + ' letters long. Try again.')
                        else:
                            print('Here is your hint:')
                            for i in range(secretNum):
                                if(guess[i] == secret[i]):
                                    print(guess[i].upper() + ' ', end='')
                                elif(guess[i] in secret):
                                    print(guess[i].lower() + ' ', end='')
                                else: print('_ ', end='')
                        print()
                        print()
                        guess = input('Enter your guess: ')
                        guesses = guesses + 1        
                        print()
                    print('You guesed it in ' + str(guesses) + ' guesses!' )
                    play = input("Do you want to play again? (yes / no) ")

                def main():
                    # Get an odometer value in U.S. miles from the user.
                    start_miles = float(input('Please enter your starting odomoter reading: '))
                    # Get another odometer value in U.S. miles from the user.
                    end_miles = float(input('Please enter your ending odometer reading: '))
                    # Get a fuel amount in U.S. gallons from the user.
                    amount_gallons = float(input('Please enter the amount of fuel consumed (U.S. gallons): '))
                    # Call the miles_per_gallon function and store
                    # the result in a variable named mpg.
                    mpg = miles_per_gallon(start_miles, end_miles, amount_gallons)
                    # Call the lp100k_from_mpg function to convert the
                    # miles per gallon to liters per 100 kilometers and
                    # store the result in a variable named lp100k.
                    lp100k = lp100k_from_mpg(mpg)
                    # Display the results for the user to see.
                    print(f'{mpg:.1f} miles per gallon')
                    print(f'{lp100k:.2f} liters per 100 kilometers')
                    pass


                def miles_per_gallon(start_miles, end_miles, amount_gallons):
                    """Compute and return the average number of miles
                    that a vehicle traveled per gallon of fuel.
                    
                    Parameters
                        start_miles: An odometer value in miles.
                        end_miles: Another odometer value in miles.
                        amount_gallons: A fuel amount in U.S. gallons.
                    Return: Fuel efficiency in miles per gallon.
                    """
                    mpg = (end_miles - start_miles)/amount_gallons
                    return mpg


                def lp100k_from_mpg(mpg):
                    """Convert miles per gallon to liters per 100
                    kilometers and return the converted value.

                    Parameter mpg: A value in miles per gallon
                    Return: The converted value in liters per 100km.
                    """
                    lp100k = 235.215 / mpg
                    return lp100k


                # Call the main function so that
                # this program will start executing.
                main()
